&= \left|\underset{r\leftarrow{{0,1}^n}}{Pr(LSB®}=1) - \underset{k\leftarrow{K}}{Pr(LSB(G(k_1)\wedge G(k_2))}=1)\right|

&= \frac{1}{4} \end{align} $$ {% endraw %}
